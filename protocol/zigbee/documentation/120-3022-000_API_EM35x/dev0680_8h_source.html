563  DECLARE_INTERRUPT_STATE; /* Must read-modify-write so to be safe disable interrupts */ \
564  DISABLE_INTERRUPTS(); \

567  /*halGpioSetConfig(RHO_CFG, PWRUP_CFG_DFL_RHO_FOR_RHO);*/ \

570  | (PWRUP_CFG_DFL_RHO_FOR_RHO << ((RHO_GPIO & 3) * 4)); \

574  /* Shadow register state */ \

577  | (PWRUP_CFG_DFL_RHO_FOR_RHO << ((RHO_GPIO & 3) * 4)); \

583  | (PWRDN_CFG_DFL_RHO_FOR_RHO << ((RHO_GPIO & 3) * 4)); \
584  gpioOutPowerDown[RHO_GPIO >> 3] = gpioOutPowerDown[RHO_GPIO >> 3] \
