358  DECLARE_INTERRUPT_STATE; /* Must read-modify-write so to be safe disable interrupts */ \
359  DISABLE_INTERRUPTS(); \

365  | (PWRUP_CFG_DFL_RHO_FOR_RHO << ((RHO_GPIO & 3) * 4)); \

372  | (PWRUP_CFG_DFL_RHO_FOR_RHO << ((RHO_GPIO & 3) * 4)); \
